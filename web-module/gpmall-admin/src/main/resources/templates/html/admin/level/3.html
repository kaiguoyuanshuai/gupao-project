这三个菜单页面均用到了路径动画，如下示例为动画在larryms框架中的使用方式：（是不是很简单，只需和使用layui官方组件一样使用另外如果使用页面路径动画库，也可以将框架本身选项卡页切换效果在页面或任意地方选择性关闭，默认开启）

var $ = layui.$,
    larryms = layui.larryms,
    svgLoader = layui.svgLoader;
// 执行svg插件引入
// larryms.plugin('snap.svg-min');（这里svgLoader已经自动完成了svg加载所以无需通过plugin执行）
var loader = new SVGLoader(document.getElementById('loader'), {
    speedIn: 400,
    easingIn: mina.easeinout
});
loader.show();
